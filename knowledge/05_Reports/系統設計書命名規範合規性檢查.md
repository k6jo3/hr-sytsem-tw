## 2. UML類別圖

### 2.1 IAM服務 - 整體架構類別圖

```mermaid
classDiagram
    direction TB
    
    %% Controller Layer
    class HR01AuthCmdController {
        +login(LoginRequest) ResponseEntity
        +logout(LogoutRequest) ResponseEntity
        +refreshToken(RefreshTokenRequest) ResponseEntity
    }
    
    class HR01UserCmdController {
        +createUser(CreateUserRequest) ResponseEntity
        +updateUser(String, UpdateUserRequest) ResponseEntity
        +deactivateUser(String) ResponseEntity
        +resetPassword(String) ResponseEntity
    }
    
    class HR01UserQryController {
        +getUserList(UserQueryRequest) ResponseEntity
        +getUser(String) ResponseEntity
    }
    
    class HR01RoleCmdController {
        +createRole(CreateRoleRequest) ResponseEntity
        +updateRole(String, UpdateRoleRequest) ResponseEntity
        +updatePermissions(String, UpdatePermissionsRequest) ResponseEntity
    }
    
    class HR01RoleQryController {
        +getRoleList(RoleQueryRequest) ResponseEntity
        +getRole(String) ResponseEntity
    }
    
    %% Application Service Layer
    class LoginServiceImpl {
        +callApi(LoginRequest, JWTModel) LoginResponse
    }
    
    class CreateUserServiceImpl {
        +callApi(CreateUserRequest, JWTModel) CreateUserResponse
    }
    
    class GetUserListServiceImpl {
        +callApi(UserQueryRequest, JWTModel) Page~UserListResponse~
    }
    
    %% Domain Service Layer
    class UserAuthenticationDomainService {
        +authenticate(String, String) User
        +validateCredentials(User, String) boolean
    }
    
    class AccountLockingDomainService {
        +recordFailureAndCheckLock(User) boolean
        +checkAndUnlock(User) boolean
    }
    
    class PasswordValidationDomainService {
        +validate(String) boolean
        +checkHistory(User, String) boolean
    }
    
    class TokenGenerationDomainService {
        +generateAccessToken(User) String
        +generateRefreshToken(User) String
        +validateToken(String) boolean
    }
    
    %% Domain Layer - Aggregates
    class User {
        -UserId id
        -String username
        -Password password
        -Email email
        -UUID employeeId
        -UUID tenantId
        -UserStatus status
        -int failedLoginAttempts
        -LocalDateTime lockedUntil
        +authenticate(String, PasswordEncoder) void
        +changePassword(String, String, PasswordEncoder) void
        +resetPassword(String, PasswordEncoder) void
        +deactivate() void
        +activate() void
        +isLocked() boolean
    }
    
    class Role {
        -RoleId id
        -String roleName
        -String displayName
        -String description
        -UUID tenantId
        -boolean isSystemRole
        -List~Permission~ permissions
        +addPermission(Permission) void
        +removePermission(Permission) void
        +hasPermission(String) boolean
    }
    
    class Permission {
        -PermissionId id
        -String permissionCode
        -String resource
        -String action
        -String description
    }
    
    %% Value Objects
    class Password {
        -String hash
        +matches(String, PasswordEncoder) boolean
        +encode(PasswordEncoder) Password
    }
    
    class Email {
        -String value
        +getValue() String
    }
    
    class UserId {
        -UUID value
        +generate() UserId
        +getValue() UUID
    }
    
    %% Repository Interfaces
    class IUserRepository {
        <<interface>>
        +findById(UserId) User
        +findByUsername(String) User
        +save(User) void
        +existsByUsername(String) boolean
    }
    
    class IRoleRepository {
        <<interface>>
        +findById(RoleId) Role
        +findByTenantId(UUID) List~Role~
        +save(Role) void
    }
    
    %% Domain Events
    class UserCreatedEvent {
        -UUID userId
        -String username
        -String email
        -UUID tenantId
    }
    
    class UserDeactivatedEvent {
        -UUID userId
        -LocalDateTime deactivatedAt
    }
    
    class UserLockedEvent {
        -UUID userId
        -LocalDateTime lockedAt
        -String reason
    }
    
    %% Relationships
    HR01AuthCmdController --> LoginServiceImpl
    HR01UserCmdController --> CreateUserServiceImpl
    HR01UserQryController --> GetUserListServiceImpl
    
    LoginServiceImpl --> UserAuthenticationDomainService
    LoginServiceImpl --> TokenGenerationDomainService
    CreateUserServiceImpl --> PasswordValidationDomainService
    
    UserAuthenticationDomainService --> IUserRepository
    UserAuthenticationDomainService --> AccountLockingDomainService
    
    User --> Password
    User --> Email
    User --> UserId
    User ..> UserCreatedEvent : publishes
    User ..> UserDeactivatedEvent : publishes
    User ..> UserLockedEvent : publishes
    
    Role --> Permission
    Role --> RoleId
    
    IUserRepository ..> User : manages
    IRoleRepository ..> Role : manages
```
